import java.util.Random;

public class Bai1 {
    public static void main(String[] args) {
        int[] a = {1,2,3,4,5,6,7,8};
        Random rand = new Random();
        int n = a.length;

        for(int i = 0; i < n; i++){
            int j = rand.nextInt(n); 

            int temp = a[i];
            a[i] = a[j];
            a[j] = temp;
        }
        for(int i = 0; i < n; i++){
            System.out.print(a[i] + " ");
        }
    }
}
